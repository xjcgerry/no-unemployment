# 零钱兑换
有5种面值的钱【1,5,10,50,100】，分别有a,b,c,d,e张。给一个数值，问最少需要多少钱能够凑成这个数值，如果凑不成返回-1  
````
输入：
5 2 2 3 5
55
输出：
2
````
````cpp
#include <bits/stdc++.h>

using namespace std;

int main() {
	int num;
	vector<int> nums;
	vector<int> money{ 1, 5, 10, 50, 100 };
	for (int i = 0; i < 5; i++) {
		cin >> num;
		nums.push_back(num);
	}
	int sum;
	cin >> sum;
	int count = 0;
	int index = 4;
	while (index >= 0) {
		if (sum >= money[index] && nums[index] > 0) {
			sum = sum - money[index];
			nums[index]--;
			count++;
		}
		else {
			index--;
		}
	}
	if (sum == 0)
		cout << count;
	else
		cout << -1;

}
````
